function searchContent() { let searchTerm = document.getElementById('searchInput').value.toLowerCase(); let content = document.getElementById('pageContent'); let paragraphs = content.getElementsByTagName('p'); // Clear previous highlights for (let p of paragraphs) { p.innerHTML = p.innerText; } if (searchTerm.trim() === '') { return; } for (let p of paragraphs) { let originalText = p.innerText; let regex = new RegExp(searchTerm, 'gi'); let newText = originalText.replace(regex, '$&'); p.innerHTML = newText; } }
